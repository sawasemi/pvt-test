        function startQuestionnaire() {
            document.getElementById('startSection').style.display = 'none';

        function showCurrentQuestion() {
            const question = questions[currentQuestion];
            const container = document.getElementById('questionContainer');
            
            let html = `
                <div class="question">
                    <div class="question-number">è³ªå• ${currentQuestion + 1}</div>
                    <div class="question-text">${question.text}</div>
            `;

            if (question.subtext) {
                html += `<div class="sub-question"><div class="sub-text">${question.subtext}</div></div>`;
            }

            if (question.type === 'time' || question.type === 'number') {
                const step = question.step ? `step="${question.step}"` : '';
                const placeholder = question.placeholder ? `placeholder="${question.placeholder}"` : '';
                html += `
                    <div class="input-group">
                        <label>${question.label}</label>
                        <input type="${question.type}" id="${question.id}" ${step} ${placeholder} 
                               value="${answers[question.id] || ''}" onchange="saveAnswer('${question.id}', this.value)">
                    </div>
                `;
            } else if (question.type === 'textarea') {
                html += `
                    <div class="input-group">
                        <label>${question.label}</label>
                        <textarea id="${question.id}" rows="3" 
                                  onchange="saveAnswer('${question.id}', this.value)">${answers[question.id] || ''}</textarea>
                    </div>
                `;
            } else if (question.type === 'radio') {
                html += '<div class="radio-group">';
                question.options.forEach((option, index) => {
                    const selected = answers[question.id] == index ? 'selected' : '';
                    html += `
                        <div class="radio-option ${selected}" onclick="selectOption('${question.id}', ${index})">
                            <input type="radio" name="${question.id}" value="${index}" ${answers[question.id] == index ? 'checked' : ''}>
                            ${option}
                        </div>
                    `;
                });
                html += '</div>';
            }

            html += '</div>';
            container.innerHTML = html;

            updateProgress();
            updateNavigation();
        }

        function selectOption(questionId, value) {
            answers[questionId] = value;
            
            // Update visual selection
            const currentRadioGroup = document.querySelector('.radio-group');
            currentRadioGroup.querySelectorAll('.radio-option').forEach((option, index) => {
                option.classList.remove('selected');
                if (index === value) {
                    option.classList.add('selected');
                    option.querySelector('input').checked = true;
                }
            });
        }

        function saveAnswer(questionId, value) {
            answers[questionId] = value;
        }

        function nextQuestion() {
            if (currentQuestion < questions.length - 1) {
                currentQuestion++;

        }

        function previousQuestion() {
            if (currentQuestion > 0) {

            }
        }

        function calculatePSQIComponents() {
            // PSQIæ¨™æº–è¨ˆç®—å¼ã«åŸºã¥ã
            const comp = {};
            
            // Component 1: Subjective sleep quality (è³ªå•6)
            comp.comp1 = parseInt(answers.q6) || 0;
            
            // Component 2: Sleep latency (è³ªå•2 + è³ªå•5a)
            const latency = parseInt(answers.sleepLatency) || 0;
            const q5a = parseInt(answers.q5a) || 0;
            let latencyScore = 0;
            if (latency <= 15) latencyScore = 0;
            else if (latency <= 30) latencyScore = 1;
            else if (latency <= 60) latencyScore = 2;
            else latencyScore = 3;
            
            const latencySum = latencyScore + q5a;
            if (latencySum === 0) comp.comp2 = 0;
            else if (latencySum <= 2) comp.comp2 = 1;
            else if (latencySum <= 4) comp.comp2 = 2;
            else comp.comp2 = 3;
            
            // Component 3: Sleep duration (è³ªå•4)
            const sleepHours = parseFloat(answers.actualSleep) || 0;
            if (sleepHours > 7) comp.comp3 = 0;
            else if (sleepHours > 6) comp.comp3 = 1;
            else if (sleepHours > 5) comp.comp3 = 2;
            else comp.comp3 = 3;
            
            // Component 4: Sleep efficiency (è³ªå•1, 3, 4ã‹ã‚‰è¨ˆç®—)
            if (answers.bedtime && answers.wakeTime && answers.actualSleep) {
                const bedtime = parseTimeToMinutes(answers.bedtime);
                let wakeTime = parseTimeToMinutes(answers.wakeTime);
                
                // èµ·åºŠæ™‚åˆ»ãŒå°±åºŠæ™‚åˆ»ã‚ˆã‚Šæ—©ã„å ´åˆã¯ç¿Œæ—¥ã¨ã™ã‚‹
                if (wakeTime <= bedtime) {
                    wakeTime += 24 * 60; // 24æ™‚é–“ã‚’åˆ†ã§åŠ ç®—
                }
                
                const timeInBed = (wakeTime - bedtime) / 60; // æ™‚é–“ã«å¤‰æ›
                const actualSleepHours = parseFloat(answers.actualSleep) || 0;
                const efficiency = (actualSleepHours / timeInBed) * 100;
                
                if (efficiency >= 85) comp.comp4 = 0;
                else if (efficiency >= 75) comp.comp4 = 1;
                else if (efficiency >= 65) comp.comp4 = 2;
                else comp.comp4 = 3;
            } else {
                comp.comp4 = 0;
            }
            
            // Component 5: Sleep disturbances (è³ªå•5b-5i)
            const disturbances = ['q5b', 'q5c', 'q5d', 'q5e', 'q5f', 'q5g', 'q5h', 'q5i'];
            const disturbanceSum = disturbances.reduce((sum, q) => sum + (parseInt(answers[q]) || 0), 0);
            
            if (disturbanceSum === 0) comp.comp5 = 0;
            else if (disturbanceSum <= 9) comp.comp5 = 1;
            else if (disturbanceSum <= 18) comp.comp5 = 2;
            else comp.comp5 = 3;
            
            // Component 6: Sleep medication (è³ªå•7)
            comp.comp6 = parseInt(answers.q7) || 0;
            
            // Component 7: Daytime dysfunction (è³ªå•8 + è³ªå•9)
            const q8 = parseInt(answers.q8) || 0;
            const q9 = parseInt(answers.q9) || 0;
            const dysfunctionSum = q8 + q9;
            
            if (dysfunctionSum === 0) comp.comp7 = 0;
            else if (dysfunctionSum <= 2) comp.comp7 = 1;
            else if (dysfunctionSum <= 4) comp.comp7 = 2;
            else comp.comp7 = 3;
            
            return comp;
        }

        function parseTimeToMinutes(timeString) {
            const [hours, minutes] = timeString.split(':').map(Number);
            return hours * 60 + minutes;
        }
